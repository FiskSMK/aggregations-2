```sh
iconv -f Windows-1250 -t UTF-8 E-Przesylka-PlacowkiPP-csv.csv | tr ';' ',' > listaPlacowek.csv
```

```sh
sed 's/#,,/,/g' listaPlacowek.csv > lp.csv
```
